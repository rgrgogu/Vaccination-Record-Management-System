SELECT `clientID`, `lName`, `gName`, `mName`, `suffix`, `sex`, `email`, `mobileNum`, `birthday`, `age` 
FROM `client_info` 
INNER JOIN email ON (client_info.email_id = email.email_id)
INNER JOIN client_login ON (email.email_id = client_login.email_id);

SELECT email_name, password
FROM `client_login` 
INNER JOIN email ON (client_login.email_id = email.email_id);

SELECT email_name, password
FROM `client_login` 
INNER JOIN email ON (client_login.email_id = email.email_id)
WHERE email='$email' AND password='$password' LIMIT 1;

SELECT `clientID`, `lName`, `gName`, `mName`, `suffix`, `sex`, `mobileNum`, `birthday`, `age` 
FROM `client_info` 
INNER JOIN client_login ON (client_info.email_id = client_login.email_id)
INNER JOIN client_health ON (client_info.email_id = client_login.email_id)
WHERE client_info.email_id = '1';

SELECT clientID, lName, gName, mName, suffix, sex, mobileNum, email.email_name, client_login.birthday, client_login.age, client_health.ph_num, category.category_name
FROM `client_info` 
INNER JOIN client_login ON (client_info.clientID = client_login.email_id)
INNER JOIN email ON (client_login.email_id = email.email_id)
INNER JOIN client_health ON (client_info.clientID = client_health.healthID)
INNER JOIN category ON (client_health.categoryID = category.categoryID)
WHERE clientID = '1';
